#### 워드프레스 + Mariadb 동시에 컨테이너를 한 네트워크에 연결

mkdir wpmd<br/>
cd wpmd<br/>


touch compose.yaml<br/>
vi compose.yaml<br/>

>service:<br/>
>  db:<br/>
>    image: mariadb<br/>
>    command: '--default-anthentication-plugin=mysql_native_password'<br/>
>    volumes:<br/>
>      - db_data:/var/lib/mysql<br/>
>    restart: always<br/>
>    environment:<br/>
>      - MARIADB_USER=wpuser<br/>
>      - MARIADB_PASSWORD=1234<br/>
>      - MARIADB_DATABASE=wpdb<br/>
>      - MARIADB_ROOT_PASSWORD=1234<br/>
>    expose:<br/>
>      - 3306<br/>
>  wordpress:<br/>
>    image: wordpress:latest<br/>
>    ports:<br/>
>      - 80:80<br/>
>    restart: always<br/>
>    environment:<br/>
>      - WORDPRESS_DB_HOST=db<br/>
>      - WORDPRESS_DB_USER=wpuser<br/>
>      - WORDPRESS_DB_PASSWORD=1234<br/>
>      - WORDPRESS_DB_NAME=wpdb<br/>
>volumes:<br/>
>  db_data:<br/>
>:wq<br/>

docker ps<br/>
